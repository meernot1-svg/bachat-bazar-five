    39→function imgFallback(e: React.SyntheticEvent<HTMLImageElement>, seed?: string) {
    40→  const t = e.currentTarget;
    41→  if (t.dataset.fallback) return;
    42→  t.dataset.fallback = '1';
    43→  t.onerror = null;
    44→  // Fallback to picsum with a seed for consistent placeholder images
    45→  const fallbackSeed = encodeURIComponent((seed || 'product').replace(/[^a-zA-Z0-9]/g, '-').slice(0, 30));
    46→  t.src = `https://picsum.photos/seed/${fallbackSeed}/400/400`;
    47→}

    53→function CountdownTimer({ targetDate }: { targetDate: Date }) {
    54→  const [diff, setDiff] = useState(Math.max(0, targetDate.getTime() - Date.now()));
    55→  useEffect(() => { const id = setInterval(() => setDiff(Math.max(0, targetDate.getTime() - Date.now())), 1000); return () => clearInterval(id); }, [targetDate]);
    56→  const h = Math.floor(diff / 3600000); const m = Math.floor((diff % 3600000) / 60000); const s = Math.floor((diff % 60000) / 1000);
    57→  return <span className="font-mono font-bold tracking-wider">{String(h).padStart(2,'0')}:{String(m).padStart(2,'0')}:{String(s).padStart(2,'0')}</span>;
    58→}

    96→async function uploadImage(file: File): Promise<string | null> {
    97→  // First try: server upload (returns data URL for Vercel, local path for dev)
    98→  const formData = new FormData();
    99→  formData.append('file', file);
   100→  try {
   101→    const res = await fetch('/api/upload', { method: 'POST', body: formData });
   102→    if (res.ok) {
   103→      const data = await res.json();
   104→      if (data.url) return data.url;
   105→    }
   106→  } catch {
   107→    // Server upload failed — fall back to client-side data URL
   108→  }
   109→  // Fallback: convert to base64 data URL on the client (works everywhere)
   110→  try {
   111→    return await new Promise<string>((resolve, reject) => {
   112→      const reader = new FileReader();
   113→      reader.onload = () => resolve(reader.result as string);
   114→      reader.onerror = reject;
   115→      reader.readAsDataURL(file);
   116→    });
   117→  } catch {
   118→    return null;
   119→  }
   120→}

   123→interface RouteInfo { view: string; id: string; query: Record<string, string>; }
   124→function parseHash(hash: string): RouteInfo { const h = hash.replace(/^#\/?/, '') || ''; const [path, qs] = h.split('?'); const parts = path.split('/').filter(Boolean); const query: Record<string, string> = {}; if (qs) qs.split('&').forEach(p => { const [k, v] = p.split('='); if (k) query[decodeURIComponent(k)] = decodeURIComponent(v || ''); }); return { view: parts[0] || '', id: parts[1] || '', query }; }
   125→function makeHash(view: string, id?: string, query?: Record<string, string>) { let h = '#/' + view; if (id) h += '/' + id; if (query && Object.keys(query).length) h += '?' + Object.entries(query).map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&'); return h; }

   259→    if (activeSearch) { const q = activeSearch.toLowerCase(); list = list.filter(p => p.name.toLowerCase().includes(q) || p.brand.toLowerCase().includes(q)); }

   261→    if (query.trending === 'true') list = list.filter(p => p.trending);
   262→    list = list.filter(p => p.price <= priceMax);

   265→      case 'price-high': list.sort((a, b) => b.price - a.price); break;

   268→      default: list.sort((a, b) => (b.featured ? 1 : 0) - (a.featured ? 1 : 0));
   269→    }
   270→    return list;
   271→  }, [s, activeCat, activeSearch, sort, priceMax, query.featured, query.trending]);
   272→  const totalPages = Math.ceil(filtered.length / perPage);
   273→  const pageItems = filtered.slice((page - 1) * perPage, page * perPage);
