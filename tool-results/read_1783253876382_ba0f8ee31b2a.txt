    44→function CountdownTimer({ targetDate }: { targetDate: Date }) {
    45→  const [diff, setDiff] = useState(Math.max(0, targetDate.getTime() - Date.now()));
    46→  useEffect(() => { const id = setInterval(() => setDiff(Math.max(0, targetDate.getTime() - Date.now())), 1000); return () => clearInterval(id); }, [targetDate]);
    47→  const h = Math.floor(diff / 3600000); const m = Math.floor((diff % 3600000) / 60000); const s = Math.floor((diff % 60000) / 1000);
    48→  return <span className="font-mono font-bold tracking-wider">{String(h).padStart(2,'0')}:{String(m).padStart(2,'0')}:{String(s).padStart(2,'0')}</span>;
    49→}

    76→// ─── ROUTER ───
    77→interface RouteInfo { view: string; id: string; query: Record<string, string>; }
    78→function parseHash(hash: string): RouteInfo { const h = hash.replace(/^#\/?/, '') || ''; const [path, qs] = h.split('?'); const parts = path.split('/').filter(Boolean); const query: Record<string, string> = {}; if (qs) qs.split('&').forEach(p => { const [k, v] = p.split('='); if (k) query[decodeURIComponent(k)] = decodeURIComponent(v || ''); }); return { view: parts[0] || '', id: parts[1] || '', query }; }
    79→function makeHash(view: string, id?: string, query?: Record<string, string>) { let h = '#/' + view; if (id) h += '/' + id; if (query && Object.keys(query).length) h += '?' + Object.entries(query).map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&'); return h; }

   201→function ShopView({ query, onQuickView, navigate }: { query: Record<string, string>; onQuickView: (p: Product) => void; navigate: (h: string) => void }) {
   202→  const s = useStore();
   203→  const [search, setSearch] = useState(query.search || '');

   205→  const [sort, setSort] = useState('featured');
   206→  const [page, setPage] = useState(1);
   207→  const [priceMax, setPriceMax] = useState(200000);
   208→  const perPage = 12;

   210→  const activeSearch = query.search || search;
   211→  const filtered = useMemo(() => {
   212→    let list = s.allProducts();
   213→    if (activeCat !== 'all') list = list.filter(p => p.category === activeCat);
   214→    if (activeSearch) { const q = activeSearch.toLowerCase(); list = list.filter(p => p.name.toLowerCase().includes(q) || p.brand.toLowerCase().includes(q)); }

   216→    if (query.trending === 'true') list = list.filter(p => p.trending);
   217→    list = list.filter(p => p.price <= priceMax);
   218→    switch (sort) {
   219→      case 'price-low': list.sort((a, b) => a.price - b.price); break;
   220→      case 'price-high': list.sort((a, b) => b.price - a.price); break;
   221→      case 'rating': list.sort((a, b) => b.rating - a.rating); break;
   222→      case 'newest': list.sort((a, b) => (b.isNew ? 1 : 0) - (a.isNew ? 1 : 0)); break;
   223→      default: list.sort((a, b) => (b.featured ? 1 : 0) - (a.featured ? 1 : 0));
   224→    }
   225→    return list;
   226→  }, [s, activeCat, activeSearch, sort, priceMax, query.featured, query.trending]);
   227→  const totalPages = Math.ceil(filtered.length / perPage);
   228→  const pageItems = filtered.slice((page - 1) * perPage, page * perPage);

   236→          <Select value={sort} onValueChange={setSort}><SelectTrigger className="w-[150px]"><SelectValue placeholder="Sort by"/></SelectTrigger><SelectContent><SelectItem value="featured">Featured</SelectItem><SelectItem value="price-low">Price: Low to High</SelectItem><SelectItem value="price-high">Price: High to Low</SelectItem><SelectItem value="rating">Top Rated</SelectItem><SelectItem value="newest">Newest</SelectItem></SelectContent></Select>
   237→        </div>
   238→      </div>
   239→      <p className="text-sm text-muted-foreground">{filtered.length} products found</p>
