    50→function CountdownTimer({ targetDate }: { targetDate: Date }) {
    51→  const [diff, setDiff] = useState(Math.max(0, targetDate.getTime() - Date.now()));
    52→  useEffect(() => {
    53→    const id = setInterval(() => setDiff(Math.max(0, targetDate.getTime() - Date.now())), 1000);
    54→    return () => clearInterval(id);
    55→  }, [targetDate]);
    56→  const h = Math.floor(diff / 3600000);
    57→  const m = Math.floor((diff % 3600000) / 60000);
    58→  const s = Math.floor((diff % 60000) / 1000);
    59→  return (
    60→    <span className="font-mono font-bold tracking-wider">
    61→      {String(h).padStart(2,'0')}:{String(m).padStart(2,'0')}:{String(s).padStart(2,'0')}
    62→    </span>
    63→  );
    64→}

   101→interface RouteInfo {
   102→  view: string;
   103→  id: string;
   104→  query: Record<string, string>;
   105→}
   106→
   107→function parseHash(hash: string): RouteInfo {
   108→  const h = hash.replace(/^#\/?/, '') || '';
   109→  const [path, qs] = h.split('?');
   110→  const parts = path.split('/').filter(Boolean);
   111→  const query: Record<string, string> = {};
   112→  if (qs) qs.split('&').forEach(p => { const [k, v] = p.split('='); if (k) query[decodeURIComponent(k)] = decodeURIComponent(v || ''); });
   113→  const view = parts[0] || '';
   114→  const id = parts[1] || '';
   115→  return { view, id, query };
   116→}
   117→
   118→function makeHash(view: string, id?: string, query?: Record<string, string>) {
   119→  let h = '#/' + view;
   120→  if (id) h += '/' + id;
   121→  if (query && Object.keys(query).length) h += '?' + Object.entries(query).map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&');
   122→  return h;
   123→}

   390→function ShopView({ onQuickView, query }: { onQuickView: (p: Product) => void; query: Record<string, string> }) {

   395→  const [sortBy, setSortBy] = useState('featured');
   396→  const [page, setPage] = useState(1);

   404→  const filtered = useMemo(() => {
   405→    let items = [...allProducts];
   406→    if (catFilter) items = items.filter(p => p.category === catFilter);
   407→    if (query.featured) items = items.filter(p => p.featured);
   408→    if (query.trending) items = items.filter(p => p.trending);

   413→      case 'price-high': items.sort((a,b) => b.price - a.price); break;
   414→      case 'rating': items.sort((a,b) => b.rating - a.rating); break;
