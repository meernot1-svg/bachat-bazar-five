   108тЖТ   108тЖТ   108тЖТ  const h = hash.replace(/^#\/?/, '') || '';
   109тЖТ   109тЖТ   109тЖТ  const [path, qs] = h.split('?');

   111тЖТ   111тЖТ   111тЖТ  const query: Record<string, string> = {};
   112тЖТ   112тЖТ   112тЖТ  if (qs) qs.split('&').forEach(p => { const [k, v] = p.split('='); if (k) query[decodeURIComponent(k)] = decodeURIComponent(v || ''); });
   113тЖТ   113тЖТ   113тЖТ  const view = parts[0] || '';
   114тЖТ   114тЖТ   114тЖТ  const id = parts[1] || '';
   115тЖТ   115тЖТ   115тЖТ  return { view, id, query };
